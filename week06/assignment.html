Of all instances of that motif in the genome (or in one chromosome), what proportion is bound by the factor (i.e. has a peak)?

# Scan the whole chr1 for EGR1 motif instances
chr1_seq <- getSeq(BSgenome.Hsapiens.UCSC.hg38, "chr1")
chr1_seq <- DNAStringSet(chr1_seq)
names(chr1_seq) <- "chr1"

motif_chr1 <- motifmatchr::matchMotifs(motif2, subject = chr1_seq,
                                        out = "positions")[[1]]
names(motif_chr1) <- "chr1"
motif_chr1 <- as(motif_chr1, "GRanges")
motif_percentge <- sum(overlapsAny(motif_chr1, peaks_chr1)) / length(motif_chr1) * 100
cat("\nOf the", length(motif_chr1), "motif instances in chromosome 1,",
    sum(overlapsAny(motif_chr1, peaks_chr1)),
    "(", motif_percentge, "%) overlap a peak")
